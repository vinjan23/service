<script>
document.addEventListener("DOMContentLoaded", function () {

  async function fetchSnapshotInfo() {
    try {
      const response = await fetch(
        'https://snapshot.vinjan-inc.com/republic/log.json'
      );

      if (!response.ok) throw new Error("Network error");

      const data = await response.json();
      const snapshotInfo = data.snapshot_info ?? data;

      if (!snapshotInfo) throw new Error("Invalid JSON");

      document.getElementById('height').textContent =
        snapshotInfo.block_height ?? "N/A";

      document.getElementById('size').textContent =
        snapshotInfo.size ?? "N/A";

      const createdAt = new Date(snapshotInfo.created_at);
      const now = new Date();
      const timeDiffMs = now - createdAt;

      const minutesAgo = Math.floor(timeDiffMs / (1000 * 60));
      const days = Math.floor(minutesAgo / (60 * 24));
      const hours = Math.floor((minutesAgo % (60 * 24)) / 60);
      const minutes = minutesAgo % 60;

      let text = "";

      if (days > 0) {
        text = `${days} day${days > 1 ? "s" : ""}`;
        if (hours > 0) {
          text += ` and ${hours} hour${hours > 1 ? "s" : ""}`;
        }
        text += " ago";
      } else if (hours > 0) {
        text = `${hours} hour${hours > 1 ? "s" : ""} ${minutes} minute${minutes > 1 ? "s" : ""} ago`;
      } else {
        text = `${minutes} minute${minutes > 1 ? "s" : ""} ago`;
      }

      document.getElementById('last-updated').textContent = text;

    } catch (error) {
      console.error('Error fetching snapshot info:', error);
      document.getElementById('height').textContent = "Unavailable";
      document.getElementById('last-updated').textContent = "Unavailable";
      document.getElementById('size').textContent = "Unavailable";
    }
  }

  fetchSnapshotInfo();
});
</script>
