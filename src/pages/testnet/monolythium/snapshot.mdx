Update Every 12 hours


```
sudo apt install lz4 -y
```
```
sudo systemctl stop monod
```
```
cp $HOME/.mono/data/priv_validator_state.json $HOME/.mono/priv_validator_state.json.backup
```
```
monod comet unsafe-reset-all --home $HOME/.mono --keep-addr-book
```
```
curl -L https://snapshot.vinjan-inc.com/monolyth/latest.tar.lz4 | lz4 -dc - | tar -xf - -C $HOME/.mono
```
```
mv $HOME/.mono/priv_validator_state.json.backup $HOME/.mono/data/priv_validator_state.json
```
```
sudo systemctl restart monod
sudo journalctl -u monod -f -o cat
```
